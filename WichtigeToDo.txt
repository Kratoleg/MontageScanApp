using System;
using System.Threading;
using System.Threading.Tasks;

class Program
{
    static void Main(string[] args)
    {
        // Start the delayed print task
        Task.Run(async () =>
        {
            await Task.Delay(5000); // Wait for 5 seconds
            Console.WriteLine("This message is printed after 5 seconds.");
        });

        // Main program continues without waiting
        Console.WriteLine("The program did not wait for the delayed print.");
        
        // Keep the console open if necessary
        Console.ReadLine();
    }
}
